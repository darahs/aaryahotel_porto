üìã Instructions:

  1. Open https://www.aaryahotel.com/gallery-list in a new tab
  2. Open Developer Tools (Press F12)
  3. Go to the Console tab
  4. Copy and paste the JavaScript code below into the console
  5. Press Enter to run it
  6. Copy the results and use them to update your HTML files

üìù JavaScript Code to Run:

(function() {
    console.log('=== Aarya Hotel Gallery Image Extractor ===\\n');
    
    const images = document.querySelectorAll('img[src*="galleryimages"], img[data-src*="galleryimages"]');
    const imageMap = {
        exterior: [],
        rooms: {
            presidential: [],
            executive: [],
            aaryaSuite: [],
            aaryaDeluxe: []
        },
        restaurant: [],
        spa: [],
        gym: [],
        pool: [],
        jacuzzi: [],
        steam: [],
        others: []
    };
    
    images.forEach((img) => {
        const src = img.src || img.getAttribute('data-src') || img.getAttribute('srcset')?.split(' ')[0];
        if (src && src.includes('galleryimages')) {
            const alt = (img.alt || img.getAttribute('alt') || '').toLowerCase();
            const filename = src.split('/').pop();
            const imageData = { filename, url: src, alt: img.alt || '' };
            
            if (alt.includes('building') || alt.includes('exterior') || alt.includes('reception') || alt.includes('lobby') || alt.includes('corridor')) {
                imageMap.exterior.push(imageData);
            } else if (alt.includes('presidential')) {
                imageMap.rooms.presidential.push(imageData);
            } else if (alt.includes('executive')) {
                imageMap.rooms.executive.push(imageData);
            } else if (alt.includes('aarya suite') || alt.includes('heritage')) {
                imageMap.rooms.aaryaSuite.push(imageData);
            } else if (alt.includes('aarya deluxe') || alt.includes('deluxe')) {
                imageMap.rooms.aaryaDeluxe.push(imageData);
            } else if (alt.includes('restaurant') || alt.includes('breakfast')) {
                imageMap.restaurant.push(imageData);
            } else if (alt.includes('spa')) {
                imageMap.spa.push(imageData);
            } else if (alt.includes('gym')) {
                imageMap.gym.push(imageData);
            } else if (alt.includes('pool') || alt.includes('swimming')) {
                imageMap.pool.push(imageData);
            } else if (alt.includes('jacuzzi')) {
                imageMap.jacuzzi.push(imageData);
            } else if (alt.includes('steam')) {
                imageMap.steam.push(imageData);
            } else {
                imageMap.others.push(imageData);
            }
        }
    });
    
    console.log('\\n=== EXTERIOR IMAGES ===');
    imageMap.exterior.forEach(img => console.log(img.url));
    
    console.log('\\n=== PRESIDENTIAL SUITE ===');
    imageMap.rooms.presidential.forEach(img => console.log(img.url));
    
    console.log('\\n=== EXECUTIVE SUITE ===');
    imageMap.rooms.executive.forEach(img => console.log(img.url));
    
    console.log('\\n=== AARYA SUITE ===');
    imageMap.rooms.aaryaSuite.forEach(img => console.log(img.url));
    
    console.log('\\n=== AARYA DELUXE ===');
    imageMap.rooms.aaryaDeluxe.forEach(img => console.log(img.url));
    
    console.log('\\n=== RESTAURANT ===');
    imageMap.restaurant.forEach(img => console.log(img.url));
    
    console.log('\\n=== SPA ===');
    imageMap.spa.forEach(img => console.log(img.url));
    
    console.log('\\n=== GYM ===');
    imageMap.gym.forEach(img => console.log(img.url));
    
    console.log('\\n=== POOL ===');
    imageMap.pool.forEach(img => console.log(img.url));
    
    console.log('\\n=== JSON OUTPUT ===');
    console.log(JSON.stringify(imageMap, null, 2));
    
    return imageMap;
})();
